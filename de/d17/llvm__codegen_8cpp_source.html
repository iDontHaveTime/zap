61 {

64 //on newer verison of llvm module_->setTargetTriple(llvm::Triple(targetTriple));
65 std::string error;
66 const auto *target = llvm::TargetRegistry::lookupTarget(targetTriple, error);
67 if (!target)
68 {
69 llvm::errs() << "Target lookup failed: " << error << "\n";
70 return false;
71 }
72
73 llvm::TargetOptions opts;

75 llvm::Reloc::PIC_);
76 module_->setDataLayout(tm->createDataLayout());
77

83 return false;
84 }
85

87 if (tm->addPassesToEmitFile(pm, dest, nullptr,

91 return false;
92 }
93
94 // TODO: Improve handling of verifying the module.
95 bool is_broken = llvm::verifyModule(*module_, &llvm::errs());
96

98 dest.flush();

103 llvm::Type *LLVMCodeGen::toLLVMType(const zir::Type &ty)
104 {
105 switch (ty.getKind())
106 {
107 case zir::TypeKind::Void:
108 return llvm::Type::getVoidTy(ctx_);
109 case zir::TypeKind::Bool:
110 return llvm::Type::getInt1Ty(ctx_);
111 case zir::TypeKind::Char:
112 return llvm::Type::getInt8Ty(ctx_);
113 case zir::TypeKind::Int:
114 return llvm::Type::getInt64Ty(ctx_);
115 case zir::TypeKind::Float:
116 return llvm::Type::getDoubleTy(ctx_);
117 case zir::TypeKind::Pointer:
118 {
119 const auto &pt = static_cast<const zir::PointerType &>(ty);
120 return llvm::PointerType::getUnqual(toLLVMType(*pt.getBaseType()));
121 }
122 case zir::TypeKind::Enum:
123 return llvm::Type::getInt64Ty(ctx_);
124 case zir::TypeKind::Record:
125 {
126 const auto &rt = static_cast<const zir::RecordType &>(ty);
127 auto it = structCache_.find(rt.getName());
128 if (it != structCache_.end())
129 return it->second;
130
131 if (rt.getName() == "String")
132 {
133 auto *structTy = llvm::StructType::create(ctx_, rt.getName());
134 structCache_[rt.getName()] = structTy;
135 std::vector<llvm::Type *> fieldTypes;
136 fieldTypes.push_back(llvm::PointerType::getUnqual(
137 llvm::Type::getInt8Ty(ctx_)));
138 fieldTypes.push_back(llvm::Type::getInt64Ty(ctx_));
139 structTy->setBody(fieldTypes);
140 return structTy;
141 }
142
143 auto *structTy = llvm::StructType::create(ctx_, rt.getName());
144 structCache_[rt.getName()] = structTy;
145 std::vector<llvm::Type *> fieldTypes;
146 for (const auto &f : rt.getFields())
147 fieldTypes.push_back(toLLVMType(*f.type));
148 structTy->setBody(fieldTypes);
149 return structTy;
150 }
151 case zir::TypeKind::Array:
152 {
153 const auto &at = static_cast<const zir::ArrayType &>(ty);
154 return llvm::ArrayType::get(toLLVMType(*at.getBaseType()), at.getSize());
155 }
156 default:
157 break;
158 }
159 throw std::runtime_error("Unknown ZIR type: " + ty.toString());
160 }

181 void LLVMCodeGen::visit(sema::BoundRootNode &node)
182 {
183 // Declare all external functions first
184 for (const auto &extFn : node.externalFunctions)
185 {
186 auto *ft = buildFunctionType(*extFn->symbol);
187 auto *f = llvm::Function::Create(ft, llvm::Function::ExternalLinkage,
188 extFn->symbol->name, *module_);
189 size_t idx = 0;
190 for (auto &arg : f->args())
191 arg.setName(extFn->symbol->parameters[idx++]->name);
192
193 functionMap_[extFn->symbol->name] = f;
194 }
195
196 // Declare all functions second so forward calls resolve
197 for (const auto &fn : node.functions)
198 {
199 auto *ft = buildFunctionType(*fn->symbol);
200 auto *f = llvm::Function::Create(ft, llvm::Function::ExternalLinkage,
201 fn->symbol->name, *module_);
202 size_t idx = 0;
203 for (auto &arg : f->args())
204 arg.setName(fn->symbol->parameters[idx++]->name);
205
206 functionMap_[fn->symbol->name] = f;
207 }
208
209 for (const auto &fn : node.functions)
210 fn->accept(*this);
211 for (const auto &rec : node.records)
212 rec->accept(*this);
213 for (const auto &en : node.enums)
214 en->accept(*this);
215 }

217 void LLVMCodeGen::visit(sema::BoundFunctionDeclaration &node)
218 {
219 auto *fn = functionMap_.at(node.symbol->name);
220 currentFn_ = fn;
221 allocaMap_.clear();
222
223 auto *entry = llvm::BasicBlock::Create(ctx_, "entry", fn);
224 builder_.SetInsertPoint(entry);
225
226 // Spill each argument to a stack slot so we can reassign params later.
227 size_t idx = 0;
228 for (auto &arg : fn->args())
229 {
230 const auto &param = node.symbol->parameters[idx++];
231 auto *alloca = createEntryAlloca(fn, param->name, arg.getType());
232 builder_.CreateStore(&arg, alloca);
233 allocaMap_[param->name] = alloca;
234 }
235
236 node.body->accept(*this);
237
238 if (!builder_.GetInsertBlock()->getTerminator())
239 {
240 if (fn->getReturnType()->isVoidTy())
241 builder_.CreateRetVoid();
242
243 }
244
245 currentFn_ = nullptr;
246 }

303 void LLVMCodeGen::visit(sema::BoundLiteral &node)
304 {
305 if (node.type->getKind() == zir::TypeKind::Record)
306 {
307 const auto &rt = static_cast<const zir::RecordType &>(*node.type);
308 if (rt.getName() == "String")
309 {
310 std::string gname;
311 auto *ptrConst = getOrCreateGlobalString(node.value, gname);
312 auto *lenConst =
313 llvm::ConstantInt::get(llvm::Type::getInt64Ty(ctx_),
314 static_cast<uint64_t>(node.value.size()));
315
316 auto *structTy = static_cast<llvm::StructType *>(toLLVMType(*node.type));
317 std::vector<llvm::Constant *> elems;
318 elems.push_back(ptrConst);
319 elems.push_back(lenConst);
320 lastValue_ = llvm::ConstantStruct::get(structTy, elems);
321 return;
322 }
323 }
324
325 auto *ty = toLLVMType(*node.type);
326 if (ty->isIntegerTy(1))
327 {
328 lastValue_ = llvm::ConstantInt::get(ty, node.value == "true" ? 1 : 0);
329 }
330 else if (ty->isIntegerTy(8))
331 {
332 int64_t code = 0;
333 if (!node.value.empty())
334 {
335 if (node.value.size() >= 2 && node.value[0] == '\\')
336 {
337 switch (node.value[1])
338 {
339 case 'n':
340 code = '\n';
341 break;
342 case 't':
343 code = '\t';
344 break;
345 case 'r':
346 code = '\r';
347 break;
348 case '\\':
349 code = '\\';
350 break;
351 case '\'':
352 code = '\'';
353 break;
354 case '0':
355 code = '\0';
356 break;
357 default:
358 code = static_cast<unsigned char>(node.value[1]);
359 break;
360 }
361 }
362 else
363 {
364 code = static_cast<unsigned char>(node.value[0]);
365 }
366 }
367 lastValue_ = llvm::ConstantInt::get(ty, code, /*isSigned=*/false);
368 }
369 else if (ty->isIntegerTy())
370 {
371 lastValue_ =
372 llvm::ConstantInt::get(ty, std::stoll(node.value), /*isSigned=*/true);
373 }
374 else if (ty->isDoubleTy())
375 {
376 lastValue_ = llvm::ConstantFP::get(ty, std::stod(node.value));
377 }

380 lastValue_ = llvm::Constant::getNullValue(ty);
381 }

391 void LLVMCodeGen::visit(sema::BoundBinaryExpression &node)
392 {
393 node.left->accept(*this);
394 auto *lhs = lastValue_;
395 node.right->accept(*this);
396 auto *rhs = lastValue_;
397
398 bool isFloat = lhs->getType()->isDoubleTy();
399
400 if (node.op == "+")
401 lastValue_ =
402 isFloat ? builder_.CreateFAdd(lhs, rhs) : builder_.CreateAdd(lhs, rhs);
403 else if (node.op == "-")
404 lastValue_ =
405 isFloat ? builder_.CreateFSub(lhs, rhs) : builder_.CreateSub(lhs, rhs);
406 else if (node.op == "*")
407 lastValue_ =
408 isFloat ? builder_.CreateFMul(lhs, rhs) : builder_.CreateMul(lhs, rhs);
409 else if (node.op == "/")
410 lastValue_ =
411 isFloat ? builder_.CreateFDiv(lhs, rhs) : builder_.CreateSDiv(lhs, rhs);
412 else if (node.op == "==")
413 lastValue_ = isFloat ? builder_.CreateFCmpOEQ(lhs, rhs)
414 : builder_.CreateICmpEQ(lhs, rhs);
415 else if (node.op == "!=")
416 lastValue_ = isFloat ? builder_.CreateFCmpONE(lhs, rhs)
417 : builder_.CreateICmpNE(lhs, rhs);
418 else if (node.op == "<")
419 lastValue_ = isFloat ? builder_.CreateFCmpOLT(lhs, rhs)
420 : builder_.CreateICmpSLT(lhs, rhs);
421 else if (node.op == "<=")
422 lastValue_ = isFloat ? builder_.CreateFCmpOLE(lhs, rhs)
423 : builder_.CreateICmpSLE(lhs, rhs);
424 else if (node.op == ">")
425 lastValue_ = isFloat ? builder_.CreateFCmpOGT(lhs, rhs)
426 : builder_.CreateICmpSGT(lhs, rhs);
427 else if (node.op == ">=")
428 lastValue_ = isFloat ? builder_.CreateFCmpOGE(lhs, rhs)
429 : builder_.CreateICmpSGE(lhs, rhs);
430 else if (node.op == "~")
431 {
432 auto *i8Ty = llvm::Type::getInt8Ty(ctx_);
433 auto *i64Ty = llvm::Type::getInt64Ty(ctx_);
434
435 llvm::Value *lhs_ptr = nullptr;
436 llvm::Value *lhs_len = nullptr;
437 llvm::Value *rhs_ptr = nullptr;
438 llvm::Value *rhs_len = nullptr;
439
440 if (node.left->type->getKind() == zir::TypeKind::Record)
441 {
442 lhs_ptr = builder_.CreateExtractValue(lhs, {0});
443 lhs_len = builder_.CreateExtractValue(lhs, {1});
444 }
445 else if (node.left->type->getKind() == zir::TypeKind::Char)
446 {
447 auto *buf = createEntryAlloca(currentFn_, "char_buf_l", i8Ty);
448 builder_.CreateStore(lhs, buf);
449 lhs_ptr = buf;
450 lhs_len = llvm::ConstantInt::get(i64Ty, 1);
451 }
452
453 if (node.right->type->getKind() == zir::TypeKind::Record)
454 {
455 rhs_ptr = builder_.CreateExtractValue(rhs, {0});
456 rhs_len = builder_.CreateExtractValue(rhs, {1});
457 }
458 else if (node.right->type->getKind() == zir::TypeKind::Char)
459 {
460 auto *buf = createEntryAlloca(currentFn_, "char_buf_r", i8Ty);
461 builder_.CreateStore(rhs, buf);
462 rhs_ptr = buf;
463 rhs_len = llvm::ConstantInt::get(i64Ty, 1);
464 }
465
466 if (functionMap_.count("string_concat_ptrlen") == 0)
467 {
468 std::vector<llvm::Type *> params = {
469 llvm::PointerType::getUnqual(i8Ty), i64Ty,
470 llvm::PointerType::getUnqual(i8Ty), i64Ty};
471 auto *ft = llvm::FunctionType::get(llvm::PointerType::getUnqual(i8Ty), params, false);
472 auto *fn = llvm::Function::Create(ft, llvm::Function::ExternalLinkage, "string_concat_ptrlen", *module_);
473 functionMap_["string_concat_ptrlen"] = fn;
474 }
475
476 auto *concatFn = functionMap_.at("string_concat_ptrlen");
477 auto *call = builder_.CreateCall(concatFn, {lhs_ptr, lhs_len, rhs_ptr, rhs_len});
478
479 auto *sumLen = builder_.CreateAdd(lhs_len, rhs_len);
480
481 auto *structTy = static_cast<llvm::StructType *>(toLLVMType(*node.type));
482 llvm::Value *res = llvm::UndefValue::get(structTy);
483 res = builder_.CreateInsertValue(res, call, {0});
484 res = builder_.CreateInsertValue(res, sumLen, {1});
485 lastValue_ = res;
486 }
487 }

516 void LLVMCodeGen::visit(sema::BoundArrayLiteral &node)
517 {
518 auto *arrayTy = static_cast<llvm::ArrayType *>(toLLVMType(*node.type));
519 auto *elemTy = arrayTy->getElementType();
520 (void)elemTy;
521
522 std::vector<llvm::Constant *> constants;
523 bool allConstants = true;
524
525 for (const auto &expr : node.elements)
526 {
527 expr->accept(*this);
528 if (auto *c = llvm::dyn_cast<llvm::Constant>(lastValue_))
529 {
530 constants.push_back(c);
531 }
532 else
533 {
534 allConstants = false;
535 break;
536 }
537 }
538
539 if (allConstants)
540 {
541 lastValue_ = llvm::ConstantArray::get(arrayTy, constants);
542 }
543 else
544 {
545 // If not all elements are constants, allocate on stack and store
546 auto *alloca = createEntryAlloca(currentFn_, "array_lit", arrayTy);
547 for (size_t i = 0; i < node.elements.size(); ++i)
548 {
549 node.elements[i]->accept(*this);
550 auto *ptr = builder_.CreateStructGEP(arrayTy, alloca, i);
551 builder_.CreateStore(lastValue_, ptr);
552 }
553 lastValue_ = builder_.CreateLoad(arrayTy, alloca);
554 }

570 void LLVMCodeGen::visit(sema::BoundIfExpression &node)
571 {
572 if (!currentFn_)
573 throw std::runtime_error("currentFn_ is null in visit(BoundIfExpression)");
574
575 auto *thenBB = llvm::BasicBlock::Create(ctx_, "if.then", currentFn_);
576 auto *elseBB = node.elseBody
577 ? llvm::BasicBlock::Create(ctx_, "if.else", currentFn_)
578 : nullptr;
579 auto *mergeBB = llvm::BasicBlock::Create(ctx_, "if.merge", currentFn_);
580
581 if (!node.condition)
582 throw std::runtime_error("condition is null in BoundIfExpression");
583 node.condition->accept(*this);
584 auto *cond = lastValue_;
585 if (!cond)
586 throw std::runtime_error(
587 "lastValue_ is null after condition in BoundIfExpression");
588
589 if (elseBB)
590 {
591 builder_.CreateCondBr(cond, thenBB, elseBB);
592 }
593 else
594 {
595 builder_.CreateCondBr(cond, thenBB, mergeBB);
596 }
597
598 builder_.SetInsertPoint(thenBB);
599 if (node.thenBody)
600 {
601 lastValue_ = nullptr;
602 node.thenBody->accept(*this);
603 }
604 auto *thenVal = lastValue_;
605 if (!builder_.GetInsertBlock()->getTerminator())
606 {
607 builder_.CreateBr(mergeBB);
608 }
609 thenBB = builder_.GetInsertBlock();
610
611 auto *elseVal = (llvm::Value *)nullptr;
612 if (elseBB)
613 {
614 builder_.SetInsertPoint(elseBB);
615 if (node.elseBody)
616 {
617 lastValue_ = nullptr;
618 node.elseBody->accept(*this);
619 }
620 elseVal = lastValue_;
621 if (!builder_.GetInsertBlock()->getTerminator())
622 {
623 builder_.CreateBr(mergeBB);
624 }
625 elseBB = builder_.GetInsertBlock();
626 }
627
628 builder_.SetInsertPoint(mergeBB);
629 if (node.type->getKind() != zir::TypeKind::Void)
630 {
631 auto *phiType = toLLVMType(*node.type);
632 auto *phi = builder_.CreatePHI(phiType, 2, "if.res");
633
634 if (thenBB->getTerminator())
635 {
636 phi->addIncoming(thenVal ? thenVal : llvm::UndefValue::get(phiType),
637 thenBB);
638 }
639
640 if (elseBB)
641 {
642 if (elseBB->getTerminator())
643 {
644 phi->addIncoming(elseVal ? elseVal : llvm::UndefValue::get(phiType),
645 elseBB);
646 }
647 }
648 else
649 {
650 phi->addIncoming(llvm::UndefValue::get(phiType), thenBB);
651 }
652 lastValue_ = phi;
653 }

656 void LLVMCodeGen::visit(sema::BoundWhileStatement &node)
657 {
658 if (!currentFn_)
659 throw std::runtime_error(
660 "currentFn_ is null in visit(BoundWhileStatement)");
661
662 auto *condBB = llvm::BasicBlock::Create(ctx_, "while.cond", currentFn_);
663 auto *bodyBB = llvm::BasicBlock::Create(ctx_, "while.body", currentFn_);
664 auto *endBB = llvm::BasicBlock::Create(ctx_, "while.end", currentFn_);
665
666 builder_.CreateBr(condBB);
667
668 builder_.SetInsertPoint(condBB);
669 if (!node.condition)
670 throw std::runtime_error("condition is null in BoundWhileStatement");
671 node.condition->accept(*this);
672 auto *cond = lastValue_;
673 if (!cond)
674 throw std::runtime_error(
675 "lastValue_ is null after condition in BoundWhileStatement");
676 builder_.CreateCondBr(cond, bodyBB, endBB);
677
678 builder_.SetInsertPoint(bodyBB);
679 loopBBStack_.push_back({condBB, endBB});
680 if (node.body)
681 node.body->accept(*this);
682 loopBBStack_.pop_back();
683 if (!builder_.GetInsertBlock()->getTerminator())
684 {
685 builder_.CreateBr(condBB);
686 }
687
688 builder_.SetInsertPoint(endBB);
689 }
