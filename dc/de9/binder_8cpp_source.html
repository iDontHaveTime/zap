354 {
355 node.left_->accept(*this);
356 if (expressionStack_.empty())
357 return;
358 auto left = std::move(expressionStack_.top());
359 expressionStack_.pop();
360
361 node.right_->accept(*this);
362 if (expressionStack_.empty())
363 return;
364 auto right = std::move(expressionStack_.top());
365 expressionStack_.pop();
366
367 auto type = left->type;
368 if (node.op_ == "+" || node.op_ == "-" || node.op_ == "*" ||
369 node.op_ == "/")
370 {
371 if (!isNumeric(left->type) || !isNumeric(right->type))
372 {
373 error(node.span, "Operator '" + node.op_ +
374 "' cannot be applied to types '" +
375 left->type->toString() + "' and '" +
376 right->type->toString() + "'");
377 }
378 type = getPromotedType(left->type, right->type);
379 }
380 else if (node.op_ == "~")
381 {
382 auto isStringOrChar = [](std::shared_ptr<zir::Type> t) {
383 if (!t) return false;
384 if (t->getKind() == zir::TypeKind::Record)
385 {
386 return static_cast<zir::RecordType *>(t.get())->getName() == "String";
387 }
388 return t->getKind() == zir::TypeKind::Char;
389 };
390
391 if (!isStringOrChar(left->type) || !isStringOrChar(right->type))
392 {
393 error(node.span, "Operator '~' can only be applied to 'Char' and 'String' types");
394 }
395 type = std::make_shared<zir::RecordType>("String");
396 }
397 else if (node.op_ == "==" || node.op_ == "!=" || node.op_ == "<" ||
398 node.op_ == ">" || node.op_ == "<=" || node.op_ == ">=")
399 {
400 if (!canConvert(left->type, right->type) &&
401 !canConvert(right->type, left->type))
402 {
403 error(node.span, "Incompatible types for comparison: '" +
404 left->type->toString() + "' and '" +
405 right->type->toString() + "'");
406 }
407 type = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Bool);
408 }
409
410 expressionStack_.push(std::make_unique<BoundBinaryExpression>(
411 std::move(left), node.op_, std::move(right), type));
412 }

694 {

696 std::shared_ptr<zir::Type> type = nullptr;
697
698 if (symbol && symbol->getKind() == SymbolKind::Type)
699 {
700 type = symbol->type;
701 }
702 else
703 {
704 if (typeNode.typeName == "Int")

706 else if (typeNode.typeName == "Float")

709 type = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Bool);

712 else

714 }
715
716 if (typeNode.isArray)

720 {
721 size = constInt->value_;
722 }
723 type = std::make_shared<zir::ArrayType>(std::move(type), size);
724 }
725
726 if (typeNode.isPointer)
727 {

794 }

796 }
