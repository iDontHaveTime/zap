34 void BoundIRGenerator::visit(sema::BoundFunctionDeclaration &node)
35 {
36 auto symbol = node.symbol;
37 auto func =
38 std::make_unique<Function>(symbol->name, symbol->returnType->toString());
39 currentFunction_ = func.get();
40
41 auto entryBlock = std::make_unique<BasicBlock>("entry");
42 currentBlock_ = entryBlock.get();
43 currentFunction_->addBlock(std::move(entryBlock));
44
45 for (const auto &paramSymbol : symbol->parameters)
46 {
47 auto arg = std::make_shared<Argument>(paramSymbol->name, paramSymbol->type);
48 currentFunction_->arguments.push_back(arg);
49
50 // Create an alloca for the parameter to make it mutable (standard LLVM
51 // practice)
52 auto allocaReg =
53 createRegister(std::make_shared<PointerType>(paramSymbol->type));
54 currentBlock_->addInstruction(
55 std::make_unique<AllocaInst>(allocaReg, paramSymbol->type));
56 currentBlock_->addInstruction(std::make_unique<StoreInst>(arg, allocaReg));
57
58 symbolMap_[paramSymbol] = allocaReg;
59 }
60
61 if (node.body)
62 {
63 node.body->accept(*this);
64 }
65
66 if (currentBlock_ &&
67 (currentBlock_->instructions.empty() ||
68 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret))
69 {
70 if (symbol->returnType->getKind() == TypeKind::Void)
71 {
72 currentBlock_->addInstruction(std::make_unique<ReturnInst>());
73 }
74 else
75 {
76 auto dummy = std::make_shared<Constant>("0", symbol->returnType);
77 currentBlock_->addInstruction(std::make_unique<ReturnInst>(dummy));
78 }
79 }
80
81 module_->addFunction(std::move(func));
82 currentFunction_ = nullptr;
83 currentBlock_ = nullptr;
84 symbolMap_.clear();
85 }

179 void BoundIRGenerator::visit(sema::BoundBinaryExpression &node)
180 {
181 node.left->accept(*this);
182 auto left = valueStack_.top();
183 valueStack_.pop();
184
185 node.right->accept(*this);
186 auto right = valueStack_.top();
187 valueStack_.pop();
188
189 auto reg = createRegister(node.type);
190 OpCode op;
191 if (node.op == "+")
192 op = OpCode::Add;
193 else if (node.op == "-")
194 op = OpCode::Sub;
195 else if (node.op == "*")
196 op = OpCode::Mul;
197 else if (node.op == "/")
198 op = OpCode::Div;
199 else if (node.op == "==" || node.op == "!=" || node.op == "<" ||
200 node.op == ">" || node.op == "<=" || node.op == ">=")
201 op = OpCode::Cmp; // Simplified
202 else
203 op = OpCode::Add;
204
205 currentBlock_->addInstruction(
206 std::make_unique<BinaryInst>(op, reg, left, right));
207 valueStack_.push(reg);
208 }

257 void BoundIRGenerator::visit(sema::BoundIfExpression &node)
258 {
259 auto trueLabel = createBlockLabel("if.then");
260 auto falseLabel = node.elseBody ? createBlockLabel("if.else") : "";
261 auto mergeLabel = createBlockLabel("if.merge");
262
263 node.condition->accept(*this);
264 auto cond = valueStack_.top();
265 valueStack_.pop();
266
267 if (node.elseBody)
268 {
269 currentBlock_->addInstruction(
270 std::make_unique<CondBranchInst>(cond, trueLabel, falseLabel));
271 }
272 else
273 {
274 currentBlock_->addInstruction(
275 std::make_unique<CondBranchInst>(cond, trueLabel, mergeLabel));
276 }
277
278 auto thenBlock = std::make_unique<BasicBlock>(trueLabel);
279 auto *thenBlockPtr = thenBlock.get();
280 currentFunction_->addBlock(std::move(thenBlock));
281 currentBlock_ = thenBlockPtr;
282
283 if (node.thenBody)
284 node.thenBody->accept(*this);
285
286 std::shared_ptr<Value> thenVal = nullptr;
287 if (!valueStack_.empty())
288 {
289 thenVal = valueStack_.top();
290 valueStack_.pop();
291 }
292 std::string actualThenLabel = currentBlock_->label;
293
294 if (currentBlock_->instructions.empty() ||
295 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret)
296 {
297 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
298 }
299
300 std::shared_ptr<Value> elseVal = nullptr;
301 std::string actualElseLabel = "";
302 if (node.elseBody)
303 {
304 auto elseBlock = std::make_unique<BasicBlock>(falseLabel);
305 auto *elseBlockPtr = elseBlock.get();
306 currentFunction_->addBlock(std::move(elseBlock));
307 currentBlock_ = elseBlockPtr;
308 node.elseBody->accept(*this);
309
310 if (!valueStack_.empty())
311 {
312 elseVal = valueStack_.top();
313 valueStack_.pop();
314 }
315 actualElseLabel = currentBlock_->label;
316
317 if (currentBlock_->instructions.empty() ||
318 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret)
319 {
320 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
321 }
322 }
323
324 auto mergeBlock = std::make_unique<BasicBlock>(mergeLabel);
325 auto *mergeBlockPtr = mergeBlock.get();
326 currentFunction_->addBlock(std::move(mergeBlock));
327 currentBlock_ = mergeBlockPtr;
328
329 if (node.type->getKind() != TypeKind::Void)
330 {
331 auto res = createRegister(node.type);
332 std::vector<std::pair<std::string, std::shared_ptr<Value>>> incoming;
333
334 auto finalThenVal =
335 thenVal ? thenVal : std::make_shared<Constant>("undef", node.type);
336 incoming.push_back({actualThenLabel, finalThenVal});
337
338 if (node.elseBody)
339 {
340 auto finalElseVal =
341 elseVal ? elseVal : std::make_shared<Constant>("undef", node.type);
342 incoming.push_back({actualElseLabel, finalElseVal});
343 }
344 currentBlock_->addInstruction(std::make_unique<PhiInst>(res, incoming));
345 valueStack_.push(res);
346 }

349 void BoundIRGenerator::visit(sema::BoundWhileStatement &node)
350 {
351 auto condLabel = createBlockLabel("while.cond");
352 auto bodyLabel = createBlockLabel("while.body");
353 auto endLabel = createBlockLabel("while.end");
354
355 currentBlock_->addInstruction(std::make_unique<BranchInst>(condLabel));
356
357 auto condBlock = std::make_unique<BasicBlock>(condLabel);
358 auto *condBlockPtr = condBlock.get();
359 currentFunction_->addBlock(std::move(condBlock));
360 currentBlock_ = condBlockPtr;
361
362 node.condition->accept(*this);
363 auto cond = valueStack_.top();
364 valueStack_.pop();
365 currentBlock_->addInstruction(
366 std::make_unique<CondBranchInst>(cond, bodyLabel, endLabel));
367
368 auto bodyBlock = std::make_unique<BasicBlock>(bodyLabel);
369 auto *bodyBlockPtr = bodyBlock.get();
370 currentFunction_->addBlock(std::move(bodyBlock));
371 currentBlock_ = bodyBlockPtr;
372 loopLabelStack_.push_back({condLabel, endLabel});
373 node.body->accept(*this);
374 loopLabelStack_.pop_back();
375 if (currentBlock_->instructions.empty() ||
376 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret)
377 {
378 currentBlock_->addInstruction(std::make_unique<BranchInst>(condLabel));
379 }
380
381 auto endBlock = std::make_unique<BasicBlock>(endLabel);
382 auto *endBlockPtr = endBlock.get();
383 currentFunction_->addBlock(std::move(endBlock));
384 currentBlock_ = endBlockPtr;
