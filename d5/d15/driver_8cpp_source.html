73bool driver::parseArgs(int argc, char **argv) {
74 opts::ZapcOptTable table;
75 unsigned missingIndex, missingCount;
76 auto argsArr = llvm::ArrayRef<const char *>(argv, (size_t)argc).slice(1);
77
78 auto args = table.ParseArgs(argsArr, missingIndex, missingCount);
79
80 ///
81 /// --help should be a priority above --version.
82 ///
83 if (args.hasArg(opts::OPT_help)) {
84 table.printHelp(llvm::outs(), ZAP_NAME_MACRO " [options] <file>",
85 "Zap Compiler");
86 return false;
87 }
88
89 ///
90 /// --version should still have priority over regular compilation.
91 ///
92 if (args.hasArg(opts::OPT_version)) {
93 llvm::outs() << "Zap Compiler v" << zap::ZAP_VERSION << '\n';
94 return false;
95 }
96
97 if (missingCount) {
98 print_red("error: ");
99 llvm::errs() << "argument to '" << args.getArgString(missingIndex)
100 << "' is missing\n";
101 return false;
102 }
103
104 ///
105 /// Check the output type flags.
106 ///
107 bool emit_llvm = args.hasArg(opts::OPT_emitLLVM);
108 bool emit_zir = args.hasArg(opts::OPT_emitZIR);
109 bool emit_s = args.hasArg(opts::OPT_emitS);
110
111 ///
112 /// Cannot mix emit modes.
113 ///
114 if (emit_llvm && emit_zir) {
115 print_red("error: ");
116 llvm::errs() << "choosing multiple emit modes isn't allowed\n";
117 return false;
118 }
119
120 ///
121 /// If -S is present choose either textual LLVM IR or assembly.
122 /// If -S is NOT present either choose LLVM BC or continue.
123 ///
124 if (emit_s) {
125 if (emit_llvm)
126 out_type = output_type::TEXT_LLVM;
127 else
128 out_type = output_type::ASM;
129 } else if (emit_llvm)
130 out_type = output_type::LLVM;
131
132 if (emit_zir)
133 out_type = output_type::ZIR;
134
135 ///
136 /// If we are on the default emit mode and -c is present, we should not link.
137 ///
138 if (out_type == output_type::EXEC) {
139 if (args.hasArg(opts::OPT_nolink)) {
140 out_type = output_type::OBJECT;
141 }
142 }
143
144 inputs = args.getAllArgValues(opts::OPT_INPUT);
145
146 auto output_str = args.getLastArgValue(opts::OPT_output, "a.out");
147 output = std::string_view(output_str.data(), output_str.size());
148
149 implicit_output = !args.hasArg(opts::OPT_output);
150
151 if (!args.hasArg(opts::OPT_nostdlib)) {
152 objects.emplace_back(ZAPC_STDLIB_PATH);
153 }
154
155 if (!inputs.empty()) {
156 return true;
157 }
158
159 llvm::errs() << "zapc: ";
160 print_red("error: ");
161 llvm::errs() << "no input files\n";
162 return false;
163}

165bool driver::splitInputs() {
166 for (const std::string &input : get_inputs()) {
167 std::filesystem::path input_path = input;
168 std::string ext = input_path.extension().string();
169 std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
170
171 if (ext == ".zap") {
172 sources.emplace_back(std::move(input_path));
173 } else if (ext == ".a" || ext == ".o") {
174 objects.emplace_back(std::move(input_path));
175 } else {
176 print_red("error: ");
177 llvm::errs() << "unknown input type: " << input << '\n';
178 return true;
179 }
180 }
181
182 return false;
183}

185bool driver::verifyOutput() {
186 const zap::driver::output_type &emit_type = get_output_type();
187
188 bool per_file_emit = (emit_type != output_type::EXEC);
189
190 if (per_file_emit && get_inputs().size() > 1 && !is_implicit_output()) {
191 print_red("error: ");
192 llvm::errs() << "cannot specify -o with multiple input files\n";
193 return true;
194 }
195
196 if (per_file_emit && !objects.empty()) {
197 print_red("error: ");
198 llvm::errs() << "cannot use object files or archives with the selected "
199 "output mode\n";
200 return true;
201 }
202
203 if (!format_supported()) {
204 print_red("error: ");
205 llvm::errs()
206 << "chosen file output mode is not yet supported in this version\n";
207 return true;
208 }
209
210 return false;

250bool driver::compileSourceFile(const std::string &source,
251 const std::string &source_name) {
252 zap::DiagnosticEngine diagnostics(source, source_name);
253 Lexer lex(diagnostics);
254
255 auto tokens = lex.tokenize(source);
256
257 zap::Parser parser(tokens, diagnostics);
258 auto ast = parser.parse();
259
260 if (diagnostics.hadErrors()) {
261 return true;
262 }
263
264 if (!ast) {
265 print_red("error: ");
266 llvm::errs() << source_name << ": " << "failed parsing the provided file\n";
267 return true;
268 }
269
270 sema::Binder binder(diagnostics);
271 auto boundAst = binder.bind(*ast);
272
273 if (!boundAst) {
274 print_red("error: ");
275 llvm::errs() << source_name << ": " << "semantic analysis failed\n";
276 return true;
277 }
278
279 if (binary_output()) {
280 if (out_type == output_type::LLVM)
281 return true; // Not yet supported.
282
283 std::filesystem::path out_path;
284
285 if (out_type == output_type::EXEC) {
286 out_path = source_name + ".o";
287 cleanups.emplace_back(out_path);
288 } else if (out_type == output_type::OBJECT) {
289 if (implicit_output) {
290 out_path = source_name + ".o";
291 } else {
292 out_path = output;
293 }
294 }
295
296 codegen::LLVMCodeGen llvmGen;
297 llvmGen.generate(*boundAst);
298
299 if (!llvmGen.emitObjectFile(out_path)) {
300 print_red("error: ");
301 llvm::errs() << "object file emission failed\n";
302 return true;
303 }
304
305 objects.emplace_back(std::move(out_path));
306 } else {
307 std::filesystem::path out_path =
308 implicit_output ? std::filesystem::path(source_name +
309 format_fileextension(out_type))
310 : output;
311
312 std::error_code ec;
313 llvm::raw_fd_ostream ofoutput(out_path.string(), ec,
314 llvm::sys::fs::OF_None);
315
316 if (ec) {
317 print_red("error: ");
318 llvm::errs() << "couldn't open the provided file: " << output << '\n';
319 llvm::errs() << "reason: " << ec.message() << '\n';
320 return true;
321 }
322
323 if (out_type == output_type::ZIR) {
324 if (compileSourceZIR(*boundAst, ofoutput))
325 return true;
326 } else if (out_type == output_type::TEXT_LLVM) {
327 codegen::LLVMCodeGen llvmGen;
328 llvmGen.generate(*boundAst);
329 llvmGen.printIR(ofoutput);
330 } else if (out_type == output_type::ASM) {
331 ///
332 /// Not yet supported.
333 ///
334 return true;
335 } else {
336 return true;
337 }
338 }
339
340 return false;
341}

343bool driver::compile() {
344 for (const std::filesystem::path &input : sources) {
345 std::ifstream file(input, std::ios::binary | std::ios::ate);
346 if (!file) {
347 print_red("error: ");
348 llvm::errs() << "couldn't open the provided file: " << input << '\n';
349 llvm::errs() << "reason: " << strerror(errno) << '\n';
350 return true;
351 }
352
353 auto size = file.tellg();
354 std::string content(size, '\0');
355
356 if (size == 0) {
357 llvm::errs() << "warning: provided file is empty: " << input << '\n';
358 } else {
359 file.seekg(0);
360 file.read(content.data(), size);
361 }
362
363 file.close();
364
365 if (compileSourceFile(content, input))
366 return true;
367 }
368
369 return false;
370}

372bool driver::link() {

374 return false;
375
376 std::vector<std::string> args;
377
378 for (const auto &obj : objects) {
379 args.emplace_back(obj);
380 }
381
382 args.push_back("-o");
383 args.push_back(output);
384
385 std::vector<llvm::StringRef> argsllvm;
386 argsllvm.push_back("cc");
387
388 for (const auto &s : args) {
389 argsllvm.push_back(s);
390 }
391
392 int res = llvm::sys::ExecuteAndWait("/usr/bin/cc", argsllvm);
393 if (res != 0) {
394 print_red("error: ");
395 llvm::errs() << "linking failed with exit code: " << res << '\n';
396 return true;
397 }
398
399 return false;
