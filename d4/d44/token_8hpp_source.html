90
91 ///
92 /// @brief Merges two 'SourceSpan' classes.
93 /// @param start From.
94 /// @param end To.
95 /// @return Merged 'SourceSpan'.
96 ///
97 static SourceSpan merge(const SourceSpan &start, const SourceSpan &end) noexcept
98 {
99 size_t newLen = (end.offset + end.length) - start.offset;
